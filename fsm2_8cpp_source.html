157 b2_height_measurement::b2_height_measurement(token* t) : state::state(t) {
158  LOG_TRACE("")
159  dispatcher* disp = TO_DISPATCHER(singleton_mgr::get_instance(DISPATCHER_PLUGIN));
160  disp->register_listener(m_token, EVENT_SENSOR_HEIGHT_R);
161  hwaccess* hal = TO_HAL(singleton_mgr::get_instance(HAL_PLUGIN));
162  int height = hal->get_height_value();
163  m_token->set_height2(height);
164  if (TOO_SMALL_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
165  && height <= TOO_SMALL_HI + HEIGHT_SENSOR_TOLERANCE_MAX) {
166  // darf nicht passieren
167  // TODO: Eigenen Fehlerzustand ?
168  new (this) err_unexpected_token(m_token);
169  } else if ((HOLE_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
170  && height <= HOLE_HI + HEIGHT_SENSOR_TOLERANCE_MAX)
171  || (METAL_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
172  && height <= METAL_HI + HEIGHT_SENSOR_TOLERANCE_MAX)) {
173  m_token->set_is_upside_down(false);
174  } else if ((HOLE_BOTTOM_UP_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
175  && height <= HOLE_BOTTOM_UP_HI + HEIGHT_SENSOR_TOLERANCE_MAX)
176  || (METAL_BOTTOM_UP_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
177  && height <= METAL_BOTTOM_UP_HI + HEIGHT_SENSOR_TOLERANCE_MAX)) {
178  new (this) b2_token_upside_down(m_token);
179  return;
180  }
181  new (this) b2_valid_height(m_token);
182 }
