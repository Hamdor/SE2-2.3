89 b1_height_measurement::b1_height_measurement(token* t) : state::state(t) {
90  LOG_TRACE("")
91  hwaccess* hal = TO_HAL(singleton_mgr::get_instance(HAL_PLUGIN));
92  int height = hal->get_height_value();
93  m_token->set_height1(height);
94  if (TOO_SMALL_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
95  && height <= TOO_SMALL_HI + HEIGHT_SENSOR_TOLERANCE_MAX) {
96  new (this) b1_token_too_small(m_token);
97  return;
98  } else if ((HOLE_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
99  && height <= HOLE_HI + HEIGHT_SENSOR_TOLERANCE_MAX)
100  || (METAL_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
101  && height <= METAL_HI + HEIGHT_SENSOR_TOLERANCE_MAX)) {
102  m_token->set_is_upside_down(false);
103  } else if ((HOLE_BOTTOM_UP_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
104  && height <= HOLE_BOTTOM_UP_HI + HEIGHT_SENSOR_TOLERANCE_MAX)
105  || (METAL_BOTTOM_UP_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
106  && height <= METAL_BOTTOM_UP_HI + HEIGHT_SENSOR_TOLERANCE_MAX)) {
107  m_token->set_is_upside_down(true);
108  }
109  new (this) b1_valid_height(m_token);
