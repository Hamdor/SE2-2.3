195 void dispatcher::direct_call_event(hal::event_values event) {
196  dispatcher_events devent = dispatcher::map_from_event_values(m_mapping,
197  event);
198  if (devent == DISPATCHED_EVENT_MAX) {
199  return;
200  }
201  if (m_prior_listners[devent] != NULL) {
202  (m_prior_listners[devent]->*m_functions[devent])();
203  return;
204  }
205  if (m_listeners[devent].empty()) {
206  return;
207  }
208  fsm::events* obj = m_listeners[devent].front();
209  (obj->*m_functions[devent])();
210  if (!m_listeners[devent].empty() && m_listeners[devent].front() == obj) {
211  // Pruefen ob das Object immer noch in der Queue,
212  // da diese evtl geaendert worden ist von einem der States...
213  m_listeners[devent].pop();
214  }
215 }

237 void dispatcher::special_case_handling(const _pulse& buffer) {
238  // Fuer eventuelle Spezialfaelle in den Signalen
239  switch(buffer.code) {
240  case INTERRUPT:
241  switch(buffer.value.sival_int) {
242  case EVENT_ZERO:
243  break;
244  case EVENT_BUTTON_START:
245  break;
246  case EVENT_BUTTON_STOP:
247  break;
248  case EVENT_BUTTON_RESET:
249  break;
250  case EVENT_BUTTON_E_STOP: {
251  token_mgr* mgr = TO_TOKEN_MGR(singleton_mgr::get_instance(TOKEN_PLUGIN));
252  mgr->enter_safe_state();
253  } break;
254  case EVENT_SENSOR_ENTRANCE:
255  case EVENT_SENSOR_HEIGHT:
256  case EVENT_SENSOR_SWITCH:
257  case EVENT_SENSOR_SLIDE:
258  case EVENT_SENSOR_EXIT: {
259  dispatcher_events devent = dispatcher::map_from_event_values(m_mapping,
260  static_cast<event_values>(buffer.value.sival_int));
261  if (devent == DISPATCHED_EVENT_MAX) {
262  break;
263  }
264  if (!m_prior_listners[devent] && m_listeners[devent].empty()) {
265  std::stringstream ss;
266  ss << "Unexpected Token triggered event: " << buffer.value.sival_int;
267  LOG_WARNING(ss.str().c_str())
268  token_mgr* mgr = TO_TOKEN_MGR(singleton_mgr::get_instance(TOKEN_PLUGIN));
269  mgr->notify_pseudo_token_not_known_event();
270  break;
271  }
272  } break;
273  case EVENT_SENSOR_ENTRANCE_R:
274  break;
275  break;
276  case EVENT_SENSOR_HEIGHT_R:
277  break;
278  case EVENT_SENSOR_SWITCH_R:
279  break;
280  case EVENT_SENSOR_SLIDE_R:
281  break;
282  case EVENT_SENSOR_EXIT_R:
283  break;
284  case EVENT_SENSOR_VALID_HEIGHT:
285  case EVENT_UNKOWN2:
286  case EVENT_UNKOWN3:
287  break;
288  default: {
289  std::stringstream ss;
290  ss << "Unkown Interrupt Value " << buffer.value.sival_int;
291  LOG_WARNING(ss.str().c_str())
292  // Unkown Event, diese Funktion schiebt einzele
293  // Masken, um zu schauen welche Events zeitgleich
294  // ausgeloest wurden und fuehrt diese aus
295  int event = 0;
296  int mask = ISR_CONCURRENT_HANDLE_START_MASK;
297  for (int i = 0; i < ISR_USED_BITS; ++i) {
298  event = mask & buffer.value.sival_int;
299  if (event) {
300  direct_call_event(static_cast<event_values>(event));
301  }
302  mask <<= 1;
303  }
304  } break;
305  }
306  break;
307  case SERIAL:
308  switch(buffer.value.sival_int) {
309  case EVENT_SERIAL_DATA:
310  break;
311  case EVENT_SERIAL_MSG:
312  break;
313  case EVENT_SERIAL_ERR:
314  break;
315  case EVENT_SERIAL_NEXT_OK: {
316  token_mgr* mgr = TO_TOKEN_MGR(singleton_mgr::get_instance(TOKEN_PLUGIN));
317  mgr->notify_ready_for_next();
318  } break;
319  case EVENT_SERIAL_TRANSFER_FIN:
320  break;
321  case EVENT_SERIAL_E_STOPP: {
322  token_mgr* mgr = TO_TOKEN_MGR(singleton_mgr::get_instance(TOKEN_PLUGIN));
323  mgr->enter_safe_state(false);
324  } break;
325  case EVENT_SERIAL_UNK:
326  break;
327  default:
328  LOG_WARNING("Unkown Serial Interface Value")
329  break;
330  }
331  break;
332  case TIMER:
333  switch(buffer.value.sival_int) {
334  case EVENT_SEG1_HAS_TO_EXPIRE:
335  break;
336  case EVENT_SEG2_HAS_TO_EXPIRE:
337  break;
338  case EVENT_SEG3_HAS_TO_EXPIRE:
339  break;
340  case EVENT_SEG1_TOO_LATE:
341  break;
342  case EVENT_SEG2_TOO_LATE:
343  break;
344  case EVENT_SEG3_TOO_LATE:
345  break;
346  case EVENT_SLIDE_FULL_TIMEOUT:
347  break;
348  case EVENT_TURN_TOKEN_TIMEOUT:
349  break;
350  case EVENT_REMOVE_TOKEN_TIMEOUT:
351  break;
352  case EVENT_CLOSE_SWITCH_TIME:
353  break;
354  case EVENT_TRANSFER_TIMEOUT:
355  break;
356  default:
357  LOG_WARNING("Unkown Timer Value")
358  break;
359  }
360  break;
361  default:
362  LOG_WARNING("Unkown Pulse Code")
363  break;
364  }
365 }

367 #define SHUTDOWN_CODE -1
368 #define SHUTDOWN_VAL -1
369 void dispatcher::execute(void*) {
370  hwaccess* hal = TO_HAL(singleton_mgr::get_instance(HAL_PLUGIN));
371  int chid = hal->get_isr_channel();
372  struct _pulse buffer;
373  std::memset(&buffer, 0, sizeof(buffer));
374  while(!isStopped()) {
375  MsgReceivePulse(chid, &buffer, sizeof(_pulse), NULL);
376  if (buffer.code == SHUTDOWN_CODE
377  && buffer.value.sival_int == SHUTDOWN_VAL) {
378  break;
379  }
380  special_case_handling(buffer);
381  direct_call_event(static_cast<event_values>(buffer.value.sival_int));
382  }
383 }
