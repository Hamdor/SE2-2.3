242 void dispatcher::execute(void*) {
243  hwaccess* hal = TO_HAL(singleton_mgr::get_instance(HAL_PLUGIN));
244  int chid = hal->get_isr_channel();
245  struct _pulse buffer;
246  std::memset(&buffer, 0, sizeof(buffer));
247  while(!isStopped()) {
248  MsgReceivePulse(chid, &buffer, sizeof(_pulse), NULL);
249  if (buffer.code == SHUTDOWN_CODE
250  && buffer.value.sival_int == SHUTDOWN_VAL) {
251  break;
252  }
253  special_case_handling(buffer);
254  direct_call_event(static_cast<event_values>(buffer.value.sival_int));
255  }
256 }
