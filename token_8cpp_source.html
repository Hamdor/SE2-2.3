117 void token::init_internal_times(int segment) {
118  token_mgr* mgr = TO_TOKEN_MGR(singleton_mgr::get_instance(TOKEN_PLUGIN));
119  const timespec add_seg1 = { (time_t)SEGMENT1__SEC, SEGMENT1_NSEC };
120  const timespec add_seg2 = { (time_t)SEGMENT2__SEC, SEGMENT2_NSEC };
121  const timespec add_seg3 = { (time_t)SEGMENT3__SEC, SEGMENT3_NSEC };
122  if (segment == SEGMENT_1) {
123  clock_gettime(CLOCK_REALTIME, &m_timespec_seg1);
124  m_timespec_seg1 = time_utils::add(m_timespec_seg1, add_seg1);
125  } else if (segment == SEGMENT_2) {
126  clock_gettime(CLOCK_REALTIME, &m_timespec_seg2);
127  m_timespec_seg2 = time_utils::add(m_timespec_seg2, add_seg2);
128  if (mgr->is_motor_slow()) {
129  const timespec addspec = mgr->get_motor_slow_diff();
130  m_timespec_seg2 = time_utils::add(m_timespec_seg2, addspec);
131  }
132  } else if (segment == SEGMENT_3) {
133  clock_gettime(CLOCK_REALTIME, &m_timespec_seg3);
134  m_timespec_seg3 = time_utils::add(m_timespec_seg3, add_seg3);
135  if (mgr->is_motor_slow()) {
136  const timespec addspec = mgr->get_motor_slow_diff();
137  m_timespec_seg3 = time_utils::add(m_timespec_seg3, addspec);
138  }
139  } else {
140  LOG_ERROR("Unkown segment value")
141  }
142 }

175 bool token::check_internal_times(int section) {
176  timespec curspec;
177  clock_gettime(CLOCK_REALTIME, &curspec);
178  std::stringstream ss;
179  if (section == SEGMENT_1) {
180  const timespec max_offset = { (time_t)OFFSET_CHECK_TIMES_SEG1__SEC,
181  OFFSET_CHECK_TIMES_SEG1_NSEC };
182  const timespec diffspec = time_utils::diff(m_timespec_seg1, curspec);
183  ss << "Segment 1 - Sec: " << diffspec.tv_sec
184  << " Nsec: " << diffspec.tv_nsec;
185  LOG_TRACE(ss.str().c_str())
186  return time_utils::smaller_then(diffspec, max_offset);
187  } else if (section == SEGMENT_2) {
188  const timespec max_offset = { (time_t)OFFSET_CHECK_TIMES_SEG2__SEC,
189  OFFSET_CHECK_TIMES_SEG2_MSEC };
190  const timespec diffspec = time_utils::diff(m_timespec_seg2, curspec);
191  ss << "Segment 2 - Sec: " << diffspec.tv_sec
192  << " Nsec: " << diffspec.tv_nsec;
193  LOG_TRACE(ss.str().c_str())
194  return time_utils::smaller_then(diffspec, max_offset);
195  } else if (section == SEGMENT_3) {
196  const timespec max_offset = { (time_t)OFFSET_CHECK_TIMES_SEG3__SEC,
197  OFFSET_CHECK_TIMES_SEG3_MSEC };
198  const timespec diffspec = time_utils::diff(m_timespec_seg3, curspec);
199  ss << "Segment 3 - Sec: " << diffspec.tv_sec
200  << " Nsec: " << diffspec.tv_nsec;
201  LOG_TRACE(ss.str().c_str())
202  return time_utils::smaller_then(diffspec, max_offset);
203  } else {
204  return false;
205  }
206 }
