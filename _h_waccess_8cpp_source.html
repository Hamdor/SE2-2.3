156 void hwaccess::init_isr() {
157  LOG_TRACE("")
158  // Channel erstellen
159  m_isr->m_chid = ChannelCreate(0);
160  if (m_isr->m_chid == -1) {
161  LOG_ERROR("ChannelCreate() failed!")
162  return;
163  }
164  // Channel oeffnen
165  m_isr->m_coid = ConnectAttach(0, 0, m_isr->m_chid, 0, 0);
166  if (m_isr->m_coid == -1) {
167  LOG_ERROR("ConnectAttach() failed!")
168  return;
169  }
170  isr_coid = m_isr->m_coid; // globale variable setzen...
171  // die isr kennt die hal nicht
172  // Interrupts zuruecksetzen
173  m_io->outbyte(IRQ_CLEAR_REG, 0);
174  // IRQ fuer Port B und Port C aktivieren
175  m_io->outbyte(IRQ_ENABLE_REG, IRQ_ENABLE_MASK);
176  // Initialisiert event struktur auf isr pulse message
177  SIGEV_PULSE_INIT(&m_isr->m_event, m_isr->m_coid, SIGEV_PULSE_PRIO_INHERIT,
178  0 /* Dieser wert definiert von wo der puls kam */, 0);
179  // Interrupt an isr (Funktion) binden
180  m_isr->m_interruptid = InterruptAttach(IO_IRQ, isr, &m_isr->m_event,
181  sizeof(m_isr->m_event), 0);
182  if (m_isr->m_interruptid == -1) {
183  LOG_ERROR("InterruptAttach() failed!")
184  return;
185  }
186  // Globale Variable der ISR initialisieren
187  port_old = (in8(static_cast<uint16_t>(PORTB)) << 8) |
188  in8(static_cast<uint16_t>(PORTC));
189 }
