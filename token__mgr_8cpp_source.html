201 bool token_mgr::check_order(bool metal) {
202  if (m_expected_token == ALL) {
203  m_expected_token = (metal ? PLASTIC : METAL);
204  return true;
205  } else if (m_expected_token == METAL) {

210  return false;

212  } else if (m_expected_token == PLASTIC) {

216  m_expected_token = METAL;
217  return true;
218  }
219  } else {

222  return false;
