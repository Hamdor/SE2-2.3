73  int coid = ConnectAttach(0, 0, hal->get_isr_channel(), 0, 0);
74  telegram data;
75  while(!isStopped()) {
76  std::memset(&data, 0, sizeof(telegram));
77  m_interface->read(&data);
78  event_values value;
79  if (data.m_type == MSG) {
80  if (data.m_msg == B2_FREE) {
81  value = EVENT_SERIAL_NEXT_OK;
82  } else {
83  value = EVENT_SERIAL_MSG;
84  }
85  } else if (data.m_type == DATA) {
86  value = EVENT_SERIAL_DATA;
87  } else if (data.m_type == ERR) {
88  value = EVENT_SERIAL_ERR;
89  } else {
90  // unkown ...
91  value = EVENT_SERIAL_UNK;
92  }
93  {
94  lock_guard lock(m_lock);
95  m_queue.push(data);
96  m_cond.broadcast();
97  }
98  MsgSendPulse(coid, SIGEV_PULSE_PRIO_INHERIT, SERIAL, value);
99  }
100 #ifndef SIMULATION
101  ConnectDetach(coid);
102 #endif
103 }
