79  if (mgr->is_motor_slow()) {
80  // Motor laeuft langsam, Zeiten anpassen

83  const duration dur = { addspec.tv_sec, (size_t)addspec.tv_nsec / MILISEC_TO_NANOSEC };
84  hdl->add_time(idx, dur);
85  }
86  if (mgr->is_motor_stopped()) {
87  // motor ist gestoppt, timer pausieren...
88  hdl->pause_timer(idx);
89  }

126 b1_height_measurement::b1_height_measurement(token* t) : state::state(t) {
127  LOG_TRACE("")
128  dispatcher* disp =
129  TO_DISPATCHER(singleton_mgr::get_instance(DISPATCHER_PLUGIN));
130  disp->register_listener(m_token, EVENT_SENSOR_HEIGHT_R);
131  hwaccess* hal = TO_HAL(singleton_mgr::get_instance(HAL_PLUGIN));
132  int height = hal->get_height_value();
133  m_token->set_height1(height);
134  if (TOO_SMALL_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
135  && height <= TOO_SMALL_HI + HEIGHT_SENSOR_TOLERANCE_MAX) {
136  new (this) b1_token_too_small(m_token);
137  return;
138  } else if ((HOLE_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
139  && height <= HOLE_HI + HEIGHT_SENSOR_TOLERANCE_MAX)
140  || (METAL_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
141  && height <= METAL_HI + HEIGHT_SENSOR_TOLERANCE_MAX)) {
142  m_token->set_is_upside_down(false);
143  } else if ((HOLE_BOTTOM_UP_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
144  && height <= HOLE_BOTTOM_UP_HI + HEIGHT_SENSOR_TOLERANCE_MAX)
145  || (METAL_BOTTOM_UP_LOW - HEIGHT_SENSOR_TOLERANCE_MIN <= height
